    <template>
    <div class="va-tag">
        <slot></slot>
        <i
            v-if="closable"
            class="va-tag_close icon-close"
            @click.stop="closeTag"
        />
    </div>
</template>

<script>
export default {
    props: ['closable'],
    methods: {
        closeTag: function($event) {
            this.$emit('close', $event);
        }
    }
}
</script>

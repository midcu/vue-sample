# 实现代码



















 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 


<template>
    <button @click="handleClick" class="va-button">
        <slot></slot>
    </button>
</template>

<script>
export default {
    name: "va-button",
    methods: {
        handleClick(evt) {
            this.$emit('click', evt);
        }
    }
}
</script>

<style scoped>
.va-button {
    display: inline-block;
    margin-bottom: 0;
    font-weight: 400;
    text-align: center;
    touch-action: manipulation;
    cursor: pointer;
    background-image: none;
    border: 1px solid transparent;
    white-space: nowrap;
    user-select: none;
    padding: 5px 15px 6px;
    font-size: 12px;
    border-radius: 4px;
    transition: color .2s linear,
        background-color .2s linear,
        border .2s linear,
        box-shadow .2s linear;
    color: #515a6e;
    background-color: #fff;
    border-color: #dcdee2;
}
</style>
